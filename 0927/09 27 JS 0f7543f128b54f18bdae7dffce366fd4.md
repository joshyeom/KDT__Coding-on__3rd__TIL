```jsx
// [QUIZ] 변수

q1 - 2;

console.log(q1); //할당을 안해줬기 때문에 그대로 3

```

```jsx
// Q2. 다음 코드 실행시 q2 변수는 어떤 값이 될까?

q2 = q2 - 2;

console.log(q2); //할당을 해서 13

```

```jsx
// console.log() 값 여러 개 찍기
```

```jsx
// 2. number

// 연산 가능
var num = 100; // 정수 
var num2 = 0.12; // 실수 

var result = 100 - 2;
```

```jsx
// 4. null
// - 빈 값. 
// - null 타입으로 "타입이 존재"하지만 "값이 없음"
// - 값이 없다는 것을 "의도적으로 명시"할 때 사용

console.log(y);

```

```jsx
// 5. undefined
```
